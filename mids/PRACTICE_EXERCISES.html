Exercise 5: Add Items to List

Task: Add new items to the list when button is clicked

Show Solution 
const input = document.querySelector('#itemInput');
const btn = document.querySelector('#addItemBtn');
const list = document.querySelector('#itemList');

btn.addEventListener('click', () => {
    const text = input.value.trim();
    
    if(text !== '') {
        const item = document.createElement('div');
        item.className = 'item';
        item.textContent = text;
        list.appendChild(item);
        input.value = '';
    }
});

Exercise 7: Active Panel Toggle

Panel 1
Panel 2
Panel 3

Task: Only one panel should be active at a time when clicked

Show Solution 
const panels = document.querySelectorAll('.panel');

panels.forEach(panel => {
    panel.addEventListener('click', () => {
        // Remove active from all
        panels.forEach(p => p.classList.remove('active'));
        // Add active to clicked one
        panel.classList.add('active');
    });
});

Exercise 8: Todo List (Add & Complete)

Task: Add todos and click them to mark as completed

Show Solution 
const input = document.querySelector('#todoInput');
const btn = document.querySelector('#addTodoBtn');
const list = document.querySelector('#todoList');

btn.addEventListener('click', () => {
    const text = input.value.trim();
    
    if(text !== '') {
        const todo = document.createElement('div');
        todo.className = 'todo-item';
        todo.textContent = text;
        
        // Click to toggle completed
        todo.addEventListener('click', () => {
            todo.classList.toggle('completed');
        });
        
        list.appendChild(todo);
        input.value = '';
    }
});

Exercise 10: Form Submit

Task: Prevent page reload and show greeting with the name entered

Show Solution 
const form = document.querySelector('#myForm');
const input = document.querySelector('#formInput');
const greeting = document.querySelector('#greeting');

form.addEventListener('submit', (e) => {
    e.preventDefault(); // IMPORTANT!
    
    const name = input.value;
    greeting.textContent = `Hello, ${name}!`;
    greeting.style.color = '#4CAF50';
    greeting.style.fontSize = '20px';
    
    input.value = '';
});
